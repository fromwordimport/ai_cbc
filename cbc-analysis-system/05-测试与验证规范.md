### 1.1 测试金字塔

```
                    ┌─────────────┐
                    │   E2E 测试   │  ← 完整分析流程验证
                    │  (少量)      │
                    ├─────────────┤
                    │  集成测试    │  ← 模块间协作
                    │  (中等)      │
                    ├─────────────┤
                    │  单元测试    │  ← 函数/方法级别
                    │  (大量)      │
                    └─────────────┘
```

| 测试类型 | 目标 | 频率 | 工具 |

```python
import numpy as np
import pandas as pd

class SyntheticDataGenerator:
    """
    合成 CBC 数据生成器
    
    生成已知参数的模拟数据，用于验证模型实现的正确性
    """
    
    def __init__(self, seed: int = 42):
        self.rng = np.random.default_rng(seed)
    
    def generate(
        self,
        n_resp: int = 500,
        n_tasks: int = 12,
        n_alts: int = 3,
        attributes: dict = None,
        true_mu: np.ndarray = None,
        true_sigma: np.ndarray = None,
        none_option: bool = True
    ) -> tuple[pd.DataFrame, dict]:
        """
        生成合成数据
        
        Parameters
        ----------
        n_resp : int
            受访者数量
        n_tasks : int
            每人选择集数量
        n_alts : int
            每集选项数
        attributes : dict
            属性定义
            {
                "brand": {"type": "categorical", "levels": ["A", "B", "C"]},
                "price": {"type": "continuous", "range": [100, 500]}
            }
        true_mu : np.ndarray
            真实总体均值
        true_sigma : np.ndarray
            真实总体标准差
        none_option : bool
            是否包含"都不选"
        
        Returns
        -------
        data : pd.DataFrame
            长格式数据
        true_params : dict
            真实参数（用于验证）
        """
        if attributes is None:
            attributes = {
                "brand": {"type": "categorical", "levels": ["A", "B", "C"]},
                "price": {"type": "continuous", "range": [100, 500]}
            }
        
        n_features = self._count_features(attributes)
        
        # 默认真实参数
        if true_mu is None:
            true_mu = np.array([0.5, -0.3, -0.002])  # brand_A, brand_B, price
        if true_sigma is None:
            true_sigma = np.array([0.8, 0.6, 0.001])
        
        # 生成个体系数
        individual_betas = []
        for _ in range(n_resp):
            beta = self.rng.normal(true_mu, true_sigma)
            individual_betas.append(beta)
        individual_betas = np.array(individual_betas)
        
        # 生成选择数据
        rows = []
        for resp_idx in range(n_resp):
            beta = individual_betas[resp_idx]
            
            for task_idx in range(n_tasks):
                # 生成选项
                alts = self._generate_alternatives(attributes, n_alts)
                
                # 计算效用
                utilities = []
                for alt in alts:
                    X = self._encode_alternative(alt, attributes)
                    util = np.dot(X, beta)
                    utilities.append(util)
                
                # 添加 Gumbel 随机项并选择
                utilities = np.array(utilities)
                gumbel = self.rng.gumbel(0, 1, n_alts)
                chosen_idx = np.argmax(utilities + gumbel)
                
                # 构建数据行
                for alt_idx, alt in enumerate(alts):
                    row = {
                        "resp_id": f"resp_{resp_idx:03d}",
                        "resp_index": resp_idx,
                        "task_id": task_idx + 1,
                        "task_index": task_idx,
                        "alt_id": alt_idx,
                        "choice": 1 if alt_idx == chosen_idx else 0
                    }
                    
                    # 添加属性值
                    for attr, value in alt.items():
                        row[attr] = value
                    
                    rows.append(row)
        
        data = pd.DataFrame(rows)
        
        true_params = {
            "mu": true_mu,
            "sigma": true_sigma,
            "individual_betas": individual_betas
        }
        
        return data, true_params
    
    def _count_features(self, attributes: dict) -> int:
        """计算特征维度"""
        n = 0
        for attr_spec in attributes.values():
            if attr_spec["type"] == "categorical":
                n += len(attr_spec["levels"]) - 1
            else:
                n += 1
        return n
    
    def _generate_alternatives(self, attributes: dict, n_alts: int) -> list[dict]:
        """生成随机选项"""
        alts = []
        for _ in range(n_alts):
            alt = {}
            for attr_name, attr_spec in attributes.items():
                if attr_spec["type"] == "categorical":
                    alt[attr_name] = self.rng.choice(attr_spec["levels"])
                else:
                    low, high = attr_spec["range"]
                    alt[attr_name] = self.rng.integers(low, high)
            alts.append(alt)
        return alts
    
    def _encode_alternative(self, alt: dict, attributes: dict) -> np.ndarray:
        """编码选项为特征向量"""
        features = []
        
        for attr_name, attr_spec in attributes.items():
            value = alt[attr_name]
            
            if attr_spec["type"] == "categorical":
                levels = attr_spec["levels"]
                idx = levels.index(value)
                n_levels = len(levels)
                
                # Effects coding
                for i in range(n_levels - 1):
                    if i == idx:
                        features.append(1)
                    elif idx == n_levels - 1:
                        features.append(-1)
                    else:
                        features.append(0)
            else:
                features.append(value)
        
        return np.array(features)
```

```python
import pytest
import numpy as np
from numpy.testing import assert_allclose

class TestModelAccuracy:
    """模型精度测试"""
    
    @pytest.fixture(scope="class")
    def synthetic_data(self):
        """生成合成数据"""
        generator = SyntheticDataGenerator(seed=42)
        data, true_params = generator.generate(
            n_resp=200,
            n_tasks=12,
            n_alts=3,
            true_mu=np.array([0.5, -0.3, -0.002]),
            true_sigma=np.array([0.8, 0.6, 0.0005])
        )
        return data, true_params
    
    def test_mnl_coefficient_recovery(self, synthetic_data):
        """测试 MNL 系数恢复精度"""
        data, true_params = synthetic_data
        
        # 拟合 MNL
        model = MNLModel(data, feature_cols=["brand_0", "brand_1", "price"])
        model.fit()
        
        # 提取估计系数
        coefs = model.get_coefficients()["coef"].values
        true_mu = true_params["mu"]
        
        # MNL 估计的是总体均值，应与 true_mu 接近
        # 允许 20% 的偏差（MNL 是聚合估计，有偏是正常的）
        relative_error = np.abs(coefs - true_mu) / np.abs(true_mu)
        
        assert np.mean(relative_error) < 0.3, \
            f"MNL 系数恢复误差过大: {relative_error}"
    
    def test_hb_coefficient_recovery(self, synthetic_data):
        """测试 HB 系数恢复精度"""
        data, true_params = synthetic_data
        
        # 拟合 HB（减少采样以加速测试）
        model = HBModel(data, feature_cols=["brand_0", "brand_1", "price"])
        model.build_model()
        trace = model.fit(n_draws=500, n_tune=500, n_chains=2)
        
        # 检查收敛
        diagnostics = model.convergence_diagnostics()
        assert diagnostics["converged"], "HB 模型未收敛"
        
        # 提取总体均值估计
        mu_posterior = trace.posterior["mu"].mean(dim=["chain", "draw"]).values
        true_mu = true_params["mu"]
        
        # HB 应更精确地恢复 true_mu
        relative_error = np.abs(mu_posterior - true_mu) / np.abs(true_mu)
        
        assert np.mean(relative_error) < 0.15, \
            f"HB 系数恢复误差过大: {relative_error}"
    
    def test_hb_sigma_recovery(self, synthetic_data):
        """测试 HB 标准差恢复"""
        data, true_params = synthetic_data
        
        model = HBModel(data, feature_cols=["brand_0", "brand_1", "price"])
        model.build_model()
        trace = model.fit(n_draws=500, n_tune=500, n_chains=2)
        
        # 提取 sigma 估计
        sigma_posterior = trace.posterior["sigma"].mean(dim=["chain", "draw"]).values
        true_sigma = true_params["sigma"]
        
        relative_error = np.abs(sigma_posterior - true_sigma) / true_sigma
        
        assert np.mean(relative_error) < 0.3, \
            f"Sigma 恢复误差过大: {relative_error}"
```
