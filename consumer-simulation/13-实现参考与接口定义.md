```python
class RequestHandler:
    """
    接收外部请求，解析为内部任务描述。
    """

    def parse_request(self, raw_request: dict) -> GenerationTask:
        """
        输入: 原始请求（来自API/CLI/消息队列）
        输出: 标准化的生成任务
        """
        # 伪代码
        validate_request_schema(raw_request)
        
        return GenerationTask(
            task_id=generate_uuid(),
            type=raw_request["type"],  # BATCH_PROFILE | SINGLE_SIMULATION
            count=raw_request.get("count", 1),
            scenario=raw_request["scenario"],
            diversity_requirements=raw_request.get("diversity", {}),
            quality_gate=raw_request.get("quality_gate", DEFAULT_QUALITY_GATE),
            created_at=now()
        )

    def validate_request_schema(self, request: dict) -> None:
        """
        校验请求格式是否合法。
        异常: InvalidRequestError
        """
        required_fields = ["type", "scenario"]
        for field in required_fields:
            if field not in request:
                raise InvalidRequestError(f"Missing required field: {field}")
```

```python
class SeedGenerator:
    """
    基于标签体系生成种子组合，确保张力充足。
    """

    def __init__(self, tag_system: TagSystem, tension_rules: list):
        self.tag_system = tag_system
        self.tension_rules = tension_rules

    def generate_seed(self, scenario: str, constraints: dict = None) -> Seed:
        """
        输入: 业务场景 + 可选约束
        输出: 种子组合 + 张力评分
        """
        # 伪代码
        for attempt in range(MAX_SEED_ATTEMPTS):
            life_stage = sample_life_stage(scenario)
            anxiety = sample_anxiety(life_stage)
            income = sample_income(life_stage)
            
            tension_score = self.calculate_tension(life_stage, anxiety, income)
            
            if TENSION_MIN <= tension_score <= TENSION_MAX:
                return Seed(
                    life_stage=life_stage,
                    anxiety=anxiety,
                    income=income,
                    tension_score=tension_score,
                    scenario=scenario
                )
        
        raise SeedGenerationError("Failed to generate valid seed after max attempts")

    def calculate_tension(self, life_stage: str, anxiety: str, income: str) -> float:
        """
        计算种子组合的矛盾指数。
        返回: 0.0 ~ 1.0
        """
        # 基于内置规则计算
        base_tension = self.tag_system.get_expected_tension(life_stage, income)
        anxiety_boost = self.tag_system.get_anxiety_tension_boost(anxiety)
        
        # 应用规则修正
        for rule in self.tension_rules:
            if rule.matches(life_stage, income):
                base_tension += rule.tension_bonus
        
        return clamp(base_tension + anxiety_boost, 0.0, 1.0)
```

### 2.3 ProfileGenerator

```python
class ProfileGenerator:
    """
    调用LLM生成消费者画像。
    """

    def __init__(self, llm_client: LLMClient, prompt_templates: PromptLibrary):
        self.llm = llm_client
        self.prompts = prompt_templates

    def generate(self, seed: Seed, tag_system: TagSystem) -> Profile:
        """
        输入: 种子 + 标签体系
        输出: 结构化画像（含原始LLM输出）
        
        异常: GenerationTimeoutError, GenerationError
        """
        # 伪代码
        prompt = self.prompts.build_profile_prompt(seed, tag_system)
        
        # Layer 1: 生成骨架（低温度，确保结构稳定）
        layer1_raw = self.llm.generate(
            prompt=prompt.layer1,
            temperature=0.3,
            timeout=30
        )
        layer1 = self.parse_layer1(layer1_raw)
        
        # Layer 2-4: 生成血肉（较高温度，允许创意）
        full_prompt = prompt.with_layer1(layer1)
        full_raw = self.llm.generate(
            prompt=full_prompt,
            temperature=0.7,
            timeout=120
        )
        
        profile = self.parse_full_output(full_raw)
        profile.seed = seed
        profile.raw_output = full_raw
        
        return profile

    def apply_fix(self, profile: Profile, fix_instructions: FixInstructions) -> Profile:
        """
        根据修正指令修改画像。
        输入: 原画像 + 修正指令
        输出: 修正后的画像（新版本）
        """
        # 伪代码
        modified_profile = profile.copy()
        
        for field_path, new_value in fix_instructions.modifications:
            modified_profile.set_field(field_path, new_value)
        
        modified_profile.version = profile.version + 1
        modified_profile.fix_history.append(fix_instructions)
        
        return modified_profile
```

### 2.4 SchemaValidator

```python
class SchemaValidator:
    """
    校验输出是否符合预定义结构。
    """

    def __init__(self, schema: JSONSchema):
        self.schema = schema

    def validate(self, profile: Profile) -> SchemaValidationResult:
        """
        输入: 画像对象
        输出: 结构校验结果
        """
        # 伪代码
        errors = []
        
        # 检查必填字段
        for field in self.schema.required_fields:
            if not profile.has_field(field):
                errors.append(SchemaError(
                    type="MISSING_FIELD",
                    field_path=field,
                    message=f"Required field '{field}' is missing"
                ))
        
        # 检查枚举值
        for field, allowed_values in self.schema.enums.items():
            value = profile.get_field(field)
            if value and value not in allowed_values:
                errors.append(SchemaError(
                    type="INVALID_ENUM",
                    field_path=field,
                    message=f"Value '{value}' not in allowed values"
                ))
        
        # 检查数组长度
        for field, (min_len, max_len) in self.schema.array_constraints.items():
            value = profile.get_field(field)
            if value and not (min_len <= len(value) <= max_len):
                errors.append(SchemaError(
                    type="ARRAY_LENGTH_VIOLATION",
                    field_path=field,
                    message=f"Array length {len(value)} not in range [{min_len}, {max_len}]"
                ))
        
        return SchemaValidationResult(
            passed=len(errors) == 0,
            errors=errors
        )
```

```python
class LogicValidator:
    """
    校验画像的内在逻辑一致性。
    """

    def __init__(self, rules: list[LogicRule]):
        self.rules = rules

    def validate(self, profile: Profile) -> LogicValidationResult:
        """
        输入: 画像对象
        输出: 逻辑校验结果
        """
        # 伪代码
        violations = []
        
        for rule in self.rules:

                if not rule.is_satisfied(profile):
                    violations.append(LogicViolation(
                        rule_id=rule.id,
                        severity=rule.severity,
                        field_path=rule.field_path,
                        description=rule.description,
                        fix_guidance=rule.fix_guidance
                    ))
        
        return LogicValidationResult(
            passed=len(violations) == 0,
            violations=violations
        )
```

```python
class BiasDetector:
    """
    检测画像中的刻板印象和偏见。
    """

    def __init__(self, stereotype_library: StereotypeLibrary, embedding_model: EmbeddingModel):
        self.library = stereotype_library
        self.embeddings = embedding_model

    def detect(self, profile: Profile, existing_profiles: list[Profile] = None) -> BiasReport:
        """
        输入: 画像 + 已有画像库（用于多样性检查）
        输出: 偏见检测报告
        """
        # 伪代码
        flags = []
        
        # 1. 刻板印象模式匹配
        for pattern in self.library.patterns:
            similarity = self.calculate_similarity(profile, pattern)
            if similarity > STEREOTYPE_THRESHOLD:
                flags.append(BiasFlag(
                    type="STEREOTYPE",
                    severity="HIGH" if similarity > 0.8 else "MEDIUM",
                    description=f"匹配刻板印象模式: {pattern.name}",
                    similarity=similarity
                ))
        
        # 2. 多样性检查
        if existing_profiles:
            diversity_score = self.calculate_diversity(profile, existing_profiles)
            if diversity_score < DIVERSITY_MIN_ENTROPY:
                flags.append(BiasFlag(
                    type="LOW_DIVERSITY",
                    severity="MEDIUM",
                    description="与已有画像过于相似",
                    diversity_score=diversity_score
                ))
        
        # 3. 叙事简化度检查
        narrative_complexity = self.assess_narrative_complexity(profile)
        if narrative_complexity < COMPLEXITY_THRESHOLD:
            flags.append(BiasFlag(
                type="OVERSIMPLIFIED_NARRATIVE",
                severity="LOW",
                description="人物小传过于标签化，缺乏个性化演绎"
            ))
        
        return BiasReport(
            has_issues=len(flags) > 0,
            flags=flags,
            overall_risk="HIGH" if any(f.severity == "HIGH" for f in flags) else "MEDIUM" if flags else "LOW"
        )

    def calculate_similarity(self, profile: Profile, pattern: StereotypePattern) -> float:
        """
        计算画像与刻板印象模式的语义相似度。
        """
        profile_embedding = self.embeddings.encode(profile.to_text())
        pattern_embedding = self.embeddings.encode(pattern.description)
        return cosine_similarity(profile_embedding, pattern_embedding)
```

```python
class AuthenticityScorer:
    """
    对画像/模拟输出进行真实感评分。
    """

    def __init__(self, rule_engine: RuleEngine, llm_reviewer: LLMClient = None):
        self.rule_engine = rule_engine
        self.llm_reviewer = llm_reviewer

    def score(self, content: Profile | SimulationRecord) -> AuthenticityScore:
        """
        输入: 画像或模拟记录
        输出: 7维度评分 + 总分
        """
        # 伪代码
        dimensions = {}
        
        # 规则引擎自动评分（可量化维度）
        dimensions["intrinsic_consistency"] = self.score_consistency(content)
        dimensions["cognitive_limitation"] = self.score_cognitive_limitation(content)
        dimensions["temporal_continuity"] = self.score_temporal_continuity(content)
        dimensions["language_naturalness"] = self.score_language_naturalness(content)
        
        # LLM辅助评分（语义维度）
        if self.llm_reviewer:
            dimensions["contextual_sensitivity"] = self.llm_score_contextual_sensitivity(content)
            dimensions["social_friction"] = self.llm_score_social_friction(content)
            dimensions["knowledge_boundary"] = self.llm_score_knowledge_boundary(content)
        
        total = sum(dimensions.values())
        
        return AuthenticityScore(
            dimensions=dimensions,
            total=total,
            max_possible=14,
            passed=total >= PASSING_THRESHOLD,
            requires_human_review=total < HUMAN_REVIEW_THRESHOLD
        )

    def llm_score_contextual_sensitivity(self, content) -> int:
        """
        使用LLM对语义维度进行评分。
        """
        prompt = f"""
        请对以下消费者画像/模拟输出的"情境敏感性"进行评分（0-2分）。
        
        评分标准：
        - 0分：场景反应与画像标签不一致，或所有场景反应雷同
        - 1分：部分一致，但存在模式化痕迹
        - 2分：不同场景下反应显著差异化，且与画像标签一致

        {content.to_text()}
        
        请只输出一个0-2的整数。
        """
        
        response = self.llm_reviewer.generate(prompt, temperature=0.1)
        return parse_int(response.strip())
```

### 2.8 FeedbackLoop

```python
class FeedbackLoop:
    """
    根据审校结果生成修正指令，管理重试流程。
    """

    def __init__(self, max_retries: int = 3):
        self.max_retries = max_retries
        self.retry_history = {}

    def create_fix_instructions(self, review_result: ReviewResult, original: Profile) -> FixInstructions:
        """
        输入: 审校结果 + 原始内容
        输出: 修正指令
        """
        # 伪代码
        modifications = []
        
        for issue in review_result.issues:
            if issue.severity in ["CRITICAL", "MAJOR"]:
                modifications.append(Modification(
                    field_path=issue.field_path,
                    current_value=original.get_field(issue.field_path),
                    target_description=issue.fix_guidance,
                    preserve_constraints=["seed", "tension_combination"]
                ))
        
        # 每次最多处理2个问题
        modifications = modifications[:2]
        
        return FixInstructions(
            target_id=original.id,
            modifications=modifications,
            strategy="targeted" if len(modifications) <= 2 else "full_regeneration",
            history=self.retry_history.get(original.id, [])
        )

    def should_retry(self, profile_id: str, new_score: float) -> bool:
        """
        判断是否应继续重试。
        """
        history = self.retry_history.get(profile_id, [])
        
        # 超过最大重试次数
        if len(history) >= self.max_retries:
            return False
        
        # 连续两次分数下降（退化检测）
        if len(history) >= 2:
            if history[-1].score < history[-2].score and new_score < history[-1].score:
                return False
        
        return True

    def record_attempt(self, profile_id: str, score: float, modifications: list):
        """
        记录重试历史。
        """
        if profile_id not in self.retry_history:
            self.retry_history[profile_id] = []
        
        self.retry_history[profile_id].append(RetryRecord(
            attempt_number=len(self.retry_history[profile_id]) + 1,
            score=score,
            modifications=modifications,
            timestamp=now()
        ))
```

### 2.10 Orchestrator

```python
class Orchestrator:
    """
    全流程编排器，管理状态机和Agent调度。
    """

    def __init__(self, config: OrchestratorConfig):
        self.config = config
        self.agents = {}
        self.state_store = StateStore()

    def run_batch_generation(self, task: GenerationTask) -> BatchResult:
        """
        批量生成画像的主流程。
        """
        # 伪代码
        results = []
        
        for i in range(task.count):
            # 1. 生成种子
            seed = self.agents["seed_gen"].generate_seed(task.scenario)
            
            # 2. 生成画像
            profile = self.agents["profile_gen"].generate(seed)
            
            # 3. 校验循环
            validated_profile = self.validation_loop(profile)
            
            if validated_profile:
                results.append(validated_profile)
            else:
                # 转人工或跳过
                results.append(self.handle_failure(profile))
        
        # 4. 生成批次报告
        report = self.agents["formatter"].format_batch_report(results)
        
        return BatchResult(profiles=results, report=report)

    def validation_loop(self, profile: Profile) -> Profile | None:
        """
        画像校验循环。
        """
        for attempt in range(self.config.max_retries + 1):
            # Schema校验
            schema_result = self.agents["schema_val"].validate(profile)
            if not schema_result.passed:
                # Schema失败直接重生成
                continue
            
            # Logic校验
            logic_result = self.agents["logic_val"].validate(profile)
            
            # Bias检测
            bias_result = self.agents["bias_det"].detect(profile)
            
            # 真实感评分
            score = self.agents["auth_score"].score(profile)
            
            # 综合判断
            if score.passed and not bias_result.has_critical_issues:
                return profile
            
            # 需要修正
            if attempt < self.config.max_retries:
                fix_instructions = self.agents["feedback"].create_fix_instructions(
                    ReviewResult.merge(logic_result, bias_result, score),
                    profile
                )
                profile = self.agents["profile_gen"].apply_fix(profile, fix_instructions)
            else:
                # 超过重试次数，转人工
                self.escalate_to_human(profile)
                return None
        
        return None

    def escalate_to_human(self, profile: Profile):
        """
        转人工审核。
        """
        self.state_store.enqueue_human_review(
            item=profile,
            reason="Exceeded max retry attempts",
            priority="HIGH"
        )
```

```python
        "HUMAN_REVIEW": {
            "human_approve": "APPROVED",
            "human_reject": "ARCHIVED"
        },
        "PASSED": {"archive": "ARCHIVED"},
        "APPROVED": {"archive": "ARCHIVED"}
    }
    
    def __init__(self):
        self.state = "CREATED"
    
    def transition(self, action: str) -> bool:
        if action in self.TRANSITIONS.get(self.state, {}):
            self.state = self.TRANSITIONS[self.state][action]
            return True
        return False
```

### 4.2 任务级状态机

```python
class TaskStateMachine:
    STATES = ["PENDING", "IN_PROGRESS", "COMPLETED", "FAILED", "CANCELLED"]
    
    def __init__(self, task_id: str):
        self.task_id = task_id
        self.state = "PENDING"
        self.subtasks = {}
    
    def start(self):
        self.state = "IN_PROGRESS"
    
    def complete_subtask(self, subtask_id: str):
        self.subtasks[subtask_id] = "COMPLETED"
        self._check_completion()
    
    def fail_subtask(self, subtask_id: str, reason: str):
        self.subtasks[subtask_id] = "FAILED"
        # 判断是否需要整体失败
        if self._is_critical_subtask(subtask_id):
            self.state = "FAILED"
    
    def _check_completion(self):
        if all(s == "COMPLETED" for s in self.subtasks.values()):
            self.state = "COMPLETED"
```

## 五、错误码定义

| 错误码 | 名称 | 说明 | 处理建议 |
|--------|------|------|---------|
| `E001` | `INVALID_REQUEST` | 请求格式不合法 | 返回400，提示缺少字段 |
| `E002` | `SEED_GENERATION_FAILED` | 种子生成失败 | 降低张力阈值重试 |
| `E003` | `GENERATION_TIMEOUT` | LLM生成超时 | 重试1次，切换备用模型 |
| `E004` | `GENERATION_ERROR` | LLM API错误 | 指数退避重试 |
| `E005` | `SCHEMA_VIOLATION` | 结构校验失败 | 要求重新生成 |
| `E006` | `LOGIC_VIOLATION` | 逻辑校验失败 | 进入FeedbackLoop |
| `E007` | `BIAS_DETECTED` | 偏见检测高危 | 强制转人工 |
| `E008` | `SCORE_TOO_LOW` | 真实感评分过低 | 进入FeedbackLoop或转人工 |
| `E009` | `MAX_RETRIES_EXCEEDED` | 超过最大重试次数 | 转人工审核 |
| `E010` | `STORAGE_ERROR` | 存储写入失败 | 重试3次后告警 |
| `E011` | `HUMAN_REVIEW_TIMEOUT` | 人工审核超时 | 自动按"最佳努力"处理 |

## 十、技术风险与应对

### 10.1 LLM调用层风险

| 风险 | 表现 | 应对 |
|------|------|------|
| **API限流** | 429 Too Many Requests | ①指数退避重试 ②多模型负载均衡 ③本地队列削峰 |
| **响应超时** | 60s+无响应 | ①流式输出+超时中断 ②降级为简化Prompt ③标记为FAILED |
| **输出截断** | 长输出被截断 | ①预估token数，超限时分段生成 ②截断检测，自动续写 |
| **模型更新** | 同一Prompt不同版本输出差异 | ①固定模型版本号 ②回归测试基线 ③输出稳定性监控 |

### 10.2 数据一致性风险

| 风险 | 场景 | 应对 |
|------|------|------|
| **MongoDB写失败** | 网络分区或主节点切换 | ①写关注 w: majority ②失败消息回队列重试 ③定期数据校验 |
| **Redis缓存不一致** | DB已更新但缓存未刷新 | ①写后删除缓存 ②设置合理TTL ③缓存穿透保护 |
| **Kafka消息丢失** | 消费者崩溃或提交失败 | ①手动提交offset ②消息幂等设计 ③死信队列 |
| **状态机漂移** | 分布式锁失效导致状态冲突 | ①乐观锁（version字段）②状态转换校验 ③定时对账 |

### 10.3 并发与性能风险

| 风险 | 触发条件 | 应对 |
|------|---------|------|
| **内存溢出** | 大量画像同时加载到Redis | ①单画像大小限制（<50KB）②LRU淘汰 ③内存告警 |
| **连接池耗尽** | LLM API并发过高 | ①连接池大小限制 ②请求排队 ③熔断降级 |
| **文件句柄泄漏** | 日志文件未关闭 | ①上下文管理器 ②定期fd检查 ③ulimit设置 |
| **CPU占满** | JSON解析或Embedding计算 | ①异步处理 ②限制并发数 ③慢任务告警 |
