### 2.2 成本熔断机制

```python
class CostCircuitBreaker:
    """成本熔断器 - 防止费用失控"""
    
    # 阈值说明（以单次标准研究 ¥500 为例）：
    # WARNING @ 80%  = ¥400 告警
    # DEGRADE @ 95%  = ¥475 自动降级
    # FUSE @ 100%    = ¥500 熔断暂停
    # EMERGENCY @ 120% = ¥600 紧急停机
    # 日度预算（¥400）和周度预算（¥2,000）按同比例计算
    FUSE_LEVELS = {
        "WARNING": {
            "threshold": 0.8,  # 预算的80%
            "action": "alert",
            "notify": ["小控", "小P"],
            "message": "成本告警：当前费用已达预算的80%，请关注"
        },
        "DEGRADE": {
            "threshold": 0.95,  # 预算的95%
            "action": "degrade",
            "notify": ["小控", "小P", "小应"],
            "message": "自动降级：费用接近预算上限，切换至轻量模型",
            "measures": [
                "画像生成切换至Haiku模型",
                "审校评分切换至规则引擎",
                "减少模拟题数（12→8题）",
                "延长缓存TTL"
            ]
        },
        "FUSE": {
            "threshold": 1.0,  # 预算的100%
            "action": "fuse",
            "notify": ["小控", "小P", "全员"],
            "message": "熔断触发：费用已达预算上限，暂停非紧急任务",
            "measures": [
                "暂停新的消费者生成",
                "暂停新的模拟任务",
                "仅保留查询类操作",
                "人工审批后恢复"
            ]
        },
        "EMERGENCY": {
            "threshold": 1.2,  # 预算的120%
            "action": "emergency_shutdown",
            "notify": ["小控", "小P", "管理层"],
            "message": "紧急停机：费用严重超标，系统已锁定",
            "measures": [
                "锁定所有LLM API调用",
                "保留只读查询",
                "启动费用审计",
                "需管理层审批解锁"
            ]
        }
    }
    
    def check_and_act(self, current_cost: float, budget: float) -> dict:
        """检查成本并执行相应动作"""
        ratio = current_cost / budget
        
        if ratio >= self.FUSE_LEVELS["EMERGENCY"]["threshold"]:
            return self._trigger("EMERGENCY")
        elif ratio >= self.FUSE_LEVELS["FUSE"]["threshold"]:
            return self._trigger("FUSE")
        elif ratio >= self.FUSE_LEVELS["DEGRADE"]["threshold"]:
            return self._trigger("DEGRADE")
        elif ratio >= self.FUSE_LEVELS["WARNING"]["threshold"]:
            return self._trigger("WARNING")
        
        return {"status": "NORMAL", "ratio": ratio}
    
    def _trigger(self, level: str) -> dict:
        """触发熔断级别"""
        config = self.FUSE_LEVELS[level]
        
        # 执行动作
        if config["action"] == "alert":
            self._send_alert(config["notify"], config["message"])
        elif config["action"] == "degrade":
            self._apply_degradation(config["measures"])
            self._send_alert(config["notify"], config["message"])
        elif config["action"] == "fuse":
            self._apply_fuse(config["measures"])
            self._send_alert(config["notify"], config["message"])
        elif config["action"] == "emergency_shutdown":
            self._emergency_shutdown(config["measures"])
            self._send_alert(config["notify"], config["message"])
        
        return {
            "status": level,
            "action": config["action"],
            "measures_applied": config["measures"],
            "notified": config["notify"]
        }
```

### 2.3 熔断后的恢复流程

```
熔断触发
   │
   ▼
系统自动执行降级/暂停
   │
   ▼
通知小控 + 小P（5分钟内）
   │
   ▼
小控分析超支原因
   │
   ├── 正常业务增长 → 申请预算追加
   │   • 小P审批 → 小控调整预算 → 系统恢复
   │
   ├── 异常消耗（如死循环、攻击）→ 修复问题
   │   • 小安协助排查 → 修复 → 小控确认 → 系统恢复
   │
   └── 配置错误 → 修正配置
       • 修正阈值 → 小控确认 → 系统恢复
   │
   ▼
人工确认后解锁
   │
   ▼
恢复后进入观察期（24小时）
   • 成本监控频率提升至每15分钟
   • 如再次触发 → 延长观察期
```

```
   │   │

   └── 成本检查
```

### 3.3 成本优化效果预估

| 优化策略 | 适用场景 | 节省比例 | 质量影响 |
|---------|---------|---------|---------|
| **模型路由** | 全局 | 30-40% | 低（审校从Sonnet→Haiku） |
| **批量调用** | 逐题模拟 | 40-50% | 无（每调用处理多题） |
| **缓存复用** | 重复画像/情境 | 20-30% | 无（命中缓存直接返回） |
| **规则引擎优先** | 审校初筛 | 50-60% | 无（规则覆盖大部分场景） |
| **分层生成** | 画像生成 | 15-20% | 无（先骨架后血肉） |

**综合优化效果**：在保持核心质量的前提下，总成本可降低 **50-65%**。

```python
class CostTracker:
    """计费追踪器 - 嵌入每个LLM调用点"""
    
    def track_llm_call(
        self,
        agent_name: str,        # "GEN" / "SIM" / "REV" / "FIX"
        model_name: str,        # "sonnet-4-6" / "haiku-4-5"
        prompt_tokens: int,
        completion_tokens: int,
        task_id: str,
        persona_id: str = None,
    ):
        """记录每次LLM调用的费用"""
        
        # 计算费用
        pricing = self._get_pricing(model_name)
        cost = (
            prompt_tokens * pricing["input_per_1k"] / 1000 +
            completion_tokens * pricing["output_per_1k"] / 1000
        )
        
        # 记录
        record = {
            "timestamp": datetime.utcnow().isoformat(),
            "task_id": task_id,
            "persona_id": persona_id,
            "agent": agent_name,
            "model": model_name,
            "prompt_tokens": prompt_tokens,
            "completion_tokens": completion_tokens,
            "total_tokens": prompt_tokens + completion_tokens,
            "cost_usd": cost,
            "cost_cny": cost * 7.2,
        }
        
        # 写入成本数据库
        self._persist(record)
        
        # 更新实时看板
        self._update_dashboard(record)
        
        # 检查熔断条件
        self._check_circuit_breaker()
        
        return record
```

### 6.3 批量优化

```python
class BatchOptimizer:
    """批量调用优化器"""
    
    def batch_choice_simulation(
        self,
        persona: Persona,
        choice_sets: list[ChoiceSet],
        batch_size: int = 3
    ) -> list[ChoiceResponse]:
        """
        将多个选择集合并为一次LLM调用
        
        优化前: 12题 × 1次调用 = 12次调用
        优化后: 12题 ÷ 3 = 4次调用
        节省: ~67% API调用次数
        """
        responses = []
        
        for i in range(0, len(choice_sets), batch_size):
            batch = choice_sets[i:i + batch_size]
            
            # 构建批量Prompt
            batch_prompt = self._build_batch_prompt(persona, batch)
            
            # 单次调用处理多题
            batch_response = self.llm.call(batch_prompt)
            
            # 解析多题结果
            parsed = self._parse_batch_response(batch_response, len(batch))
            responses.extend(parsed)
        
        return responses
```
